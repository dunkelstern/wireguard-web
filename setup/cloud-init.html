Cloud-Init

Following is an example cloud-init file for a debian host:

cloud-init.txt
 1     #cloud-config
 2
 3     ---
 4     package_update: true
 5     package_upgrade: true
 6     packages:
 7     - wireguard-tools
 8     - dnsmasq
 9     - nginx
10     - python3
11     - python3-venv
12     - python3-pip
13     - python3-certbot
14     - python3-certbot-nginx
15     - git
16     - vim-nox
17     - dnsutils
18     - wget
19     users:
20     -   groups: users
21         name: wireguard-web
22         shell: /bin/bash
23         sudo:
24         -   ALL=(ALL) NOPASSWD: /usr/bin/wg show all endpoints
25         -   ALL=(ALL) NOPASSWD: /usr/bin/wg show all latest-handshakes
26         ssh_authorized_keys:
27         - <key>
28     write_files:
29     - owner: wireguard-web:users
30       path: /home/wireguard-web/wireguard-web/.env
31       permissions: '0644'
32       content: |
33         WIREGUARD_WEB_BASE_URL=http://localhost:8000
34         WIREGUARD_STAGING_CONFIG_DIRECTORY=/tmp/wireguard-staging
35         WIREGUARD_WEB_EMAIL_HOST=localhost
36         WIREGUARD_WEB_EMAIL_PORT=25
37         WIREGUARD_WEB_EMAIL_USER=
38         WIREGUARD_WEB_EMAIL_PASSWORD=
39         WIREGUARD_WEB_EMAIL_TLS=0
40         WIREGUARD_WEB_EMAIL_BACKEND=django.core.mail.backends.console.EmailBackend
41         WIREGUARD_WEB_EMAIL_FROM=root@localhost
42         WIREGUARD_WEB_DEBUG=0
43     runcmd:
44     - "wget https://github.com/dunkelstern/wireguard-web/archive/refs/tags/0.4.0.tar.gz -O /home/wireguard-web/wireguard-web-0.4.0.tar.gz"
45     - "tar -xvz -C /home/wireguard-web -f /home/wireguard-web/wireguard-web-0.4.0.tar.gz"
46     - "su wireguard-web -c 'cd /home/wireguard-web/wireguard-web ; ./setup_wireguard_web.sh'"
47     - "systemctl enable --now wireguard-web-config.path"
